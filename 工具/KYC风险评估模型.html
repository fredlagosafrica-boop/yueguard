评分算法
# KYC 综合风险评分模型 # 公式：总分 = Σ(各因子得分 × 权重) # # 权重分配： # W_origin = 0.20 # 客群来源（权重最高） # W_id = 0.20 # 证件完整度 # W_sof = 0.25 # 资金来源风险（最高权重） # W_pep = 0.15 # PEP筛查 # W_match = 0.10 # 保额财力匹配 # W_tax = 0.10 # CRS/FATCA税务 # # 风险等级阈值： # 0 - 30 → 低风险（STD）标准流程 # 31 - 55 → 中风险（EDD）强化审查 # 56 - 80 → 高风险（HIGH）高级审查 # 81+ → 强增强（Enhanced）高管审批 # # 示例： # 内地客(15) + 证件齐全(0) + 常规收入(0) + PEP(0) + 轻微超出(10) + 双重税务(10) # = 35分 → 中风险（EDD）强化审查
